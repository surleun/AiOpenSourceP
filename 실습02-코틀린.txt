class MainActivity : AppCompatActivity() {

    private lateinit var text1: TextView
    private lateinit var text2: TextView
    private lateinit var chkAgree: CheckBox
    private lateinit var rGroup1: RadioGroup
    private lateinit var rdoDog: RadioButton
    private lateinit var rdoCat: RadioButton
    private lateinit var rdoRab: RadioButton
    private lateinit var btnOk: Button
    private lateinit var imgPet: ImageView

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)
        title = "애완동물 사진 보기"

        text1 = findViewById(R.id.Text1)
        chkAgree = findViewById(R.id.ChkAgeree)

        text2 = findViewById(R.id.Text2)
        rGroup1 = findViewById(R.id.Rgroup1)
        rdoDog = findViewById(R.id.RdoDog)
        rdoCat = findViewById(R.id.RdoCat)
        rdoRab = findViewById(R.id.RdoRab)

        btnOk = findViewById(R.id.BtnOk)
        imgPet = findViewById(R.id.ImgPet)

        chkAgree.setOnCheckedChangeListener { _, isChecked ->
            val visibility = if (isChecked) View.VISIBLE else View.INVISIBLE
            text2.visibility = visibility
            rGroup1.visibility = visibility
            btnOk.visibility = visibility
            imgPet.visibility = visibility
        }

        btnOk.setOnClickListener {
            when (rGroup1.checkedRadioButtonId) {
                R.id.RdoDog -> imgPet.setImageResource(R.drawable.dog)
                R.id.RdoCat -> imgPet.setImageResource(R.drawable.cat)
                R.id.RdoRab -> imgPet.setImageResource(R.drawable.rab)
                else -> Toast.makeText(this, "동물 먼저 선택하세요", Toast.LENGTH_SHORT).show()
            }
        }
    }
}
